Q4. Even Odd.

        
        const findEvenOdd = (N, Arr) => 
        { 
        let total = [];
        let even_sum = 0;
        let odd_sum = 0;
        for(i=0; i< N;i++){
        if (Arr[i]%2===0){
        even_sum += Arr[i]; 
        }
        else{
        odd_sum += Arr[i];
        }
        }
        total.push(even_sum);
        total.push(odd_sum);
        return total;
        };

        function Birthday_Game(arr,D ,M) {

            for (j=i;j< M+i;j++){
